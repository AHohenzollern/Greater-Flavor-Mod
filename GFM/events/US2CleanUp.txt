country_event = {
    id = 888880100
    title = "Ai US2 Cleanup"
    desc = "Ai US2 Cleanup"
    picture = "election"

    trigger = {
        ai = yes
		exists = US2
    }

    mean_time_to_happen = {
        days = 1
    }

    option = {
        name = "US2 becomes USA"
		any_country = {
			limit = {
					exists = yes
			}
			any_owned = {
				limit = { owned_by = US2 }
				secede_province = USA
			}
			any_core = {
				limit = { tag = US2 }
				add_core = USA
				remove_core = US2
			}
		}
    }
}
