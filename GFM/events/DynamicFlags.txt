    desc = "This nation will be considered a Government_Democracy with stars."

country_event = {

    id = 777770600

    title = "The 13 State Republic" # New State
    desc = "The states of the replublic now number 13, Our flag must represent this accordingly."
    picture = "ottoman"

    trigger = {
		tag = USA
		has_country_flag = Government_Democracy
		NOT = { check_variable = { which = USA_Stars_Count value = 13 }  }
		number_of_states = 1
		NOT = { number_of_states = 15 }
    }

    mean_time_to_happen = {
        days = 1
        modifier = {
            factor = 1
            USA = { exists = yes }
        }
    }

    option = {
        name = "*Stary Eyed*"
		set_variable = { which = USA_Stars_Count value = 13 }
		government = democracy_5th
        ai_chance = {
            factor = 100
        }
    }
}
